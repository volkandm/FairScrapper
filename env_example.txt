# Web Scraper Environment Configuration Example
# Copy this file to .env and update with your actual values

# API Keys (comma-separated for multiple keys)
VALID_API_KEYS=sk-demo-key-12345,sk-your-key-here

# API Host and Port Configuration
API_HOST=127.0.0.1
API_PORT=8888

# SSL/HTTPS Configuration
# Set to 'true' to enable HTTPS with self-signed certificate
SSL_ENABLED=true
SSL_CERT_FILE=./ssl/cert.pem
SSL_KEY_FILE=./ssl/key.pem

# Proxy Configuration (Optional)
# Enable/disable proxy usage
PROXY_ENABLED=true

# Multiple Proxy Configuration
# Format 1: Simple proxy URLs (no authentication)
PROXY_LIST=http://proxy1.example.com:8080,http://proxy2.example.com:8080

# Format 2: Proxy URLs with individual credentials
# PROXY_LIST=http://proxy1.example.com:8080:user1:pass1,http://proxy2.example.com:8080:user2:pass2

# Format 3: Mixed (some with credentials, some without)
# PROXY_LIST=http://proxy1.example.com:8080:user1:pass1,http://proxy2.example.com:8080,http://proxy3.example.com:8080:user3:pass3

# Format 4: SOCKS4/SOCKS5 proxy support
# PROXY_LIST=socks4://proxy1.example.com:1080,socks5://proxy2.example.com:1080,http://proxy3.example.com:8080

# Format 5: Mixed proxy types with credentials
# PROXY_LIST=socks5://proxy1.example.com:1080:user1:pass1,http://proxy2.example.com:8080:user2:pass2,socks4://proxy3.example.com:1080

# Format 6: Working free proxies (tested and verified)
# PROXY_LIST=http://66.36.234.130:1339,http://23.237.210.82:80,http://20.13.34.208:8118,http://45.167.126.1:8080,http://52.184.80.192:23128

# Proxy settings
PROXY_ROTATION_ENABLED=true
PROXY_MAX_FAILURES=3
PROXY_TEST_INTERVAL=3600

# Browser settings
HEADLESS=false
TIMEOUT=30000

# User agent
USER_AGENT="Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36"

# Default scraping settings
DEFAULT_WAIT_TIME=5000
MAX_RETRIES=3

# Stealth mode (optional): reduce bot detection on challenge pages
# Set to true to enable playwright-stealth for browser contexts
USE_STEALTH=false

# Session / challenge handling (inspired by cloudscraper)
# Force new session after this many seconds (0 = use full 30-day TTL only)
SESSION_REFRESH_INTERVAL_SEC=3600
# Auto-retry with fresh session when challenge page is detected
AUTO_REFRESH_ON_CHALLENGE=true
# Max retries with fresh session when stuck on challenge page
MAX_CHALLENGE_RETRIES=2
# Random wait range (seconds) when challenge page is detected before clicking verify
CHALLENGE_WAIT_MIN_SEC=5.0
CHALLENGE_WAIT_MAX_SEC=10.0

# Stealth: random delay before first navigation (seconds)
STEALTH_MIN_DELAY=0.5
STEALTH_MAX_DELAY=2.0

# Page ready wait: networkidle adds ~20s on many sites (analytics/websockets). Default: disabled.
PAGE_READY_USE_NETWORKIDLE=false
# Timeout for domcontentloaded and readyState complete (ms). Default 8000.
PAGE_READY_TIMEOUT_MS=8000

# Post-navigation sleep (seconds) to allow JS to run. Default 0.5. Was 2.0 before.
NAVIGATE_POST_SLEEP_SEC=0.5

# Proxy: time-based ban in seconds after a failure (0 = only use PROXY_MAX_FAILURES count)
PROXY_BAN_TIME_SEC=300

# Concurrency: max parallel scrape requests (browsers). Default 10.
MAX_CONCURRENT_SCRAPES=10
# Max queue size: reject with "Too busy, try again" when queue full. Default 100.
MAX_QUEUE_SIZE=100
# Load threshold: wait before starting new work when system load exceeds this. Default 10.
LOAD_THRESHOLD=10
